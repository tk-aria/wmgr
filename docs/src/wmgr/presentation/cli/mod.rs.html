199    pub async fn run(self) -> anyhow::Result<()> {
200        // Set up colored output
201        if !self.cli.no_color {
202            colored::control::set_override(true);
203        } else {
204            colored::control::set_override(false);
205        }
206
207        // Change directory if specified
208        if let Some(ref dir) = self.cli.directory {
209            env::set_current_dir(dir)?;
210        }
211
212        // Handle the command
213        match self.handle_command().await {
214            Ok(_) => Ok(()),
215            Err(e) => {
216                eprintln!("{} {}", "Error:".red().bold(), e);
217                exit(1);
218            }
219        }
220    }

222    async fn handle_command(&self) -> anyhow::Result<()> {
223        match &self.cli.command {
224            Commands::Init {
225                path,
226                force,
227                manifest,
228            } => {
229                self.handle_init_command(path.as_ref(), *force, *manifest)
230                    .await
231            }
232            Commands::Sync {
233                group,
234                force,
235                no_correct_branch,
236                jobs,
237            } => {
238                self.handle_sync_command(group, *force, *no_correct_branch, *jobs)
239                    .await
240            }
241            Commands::Status {
242                branch,
243                compact,
244                group,
245            } => self.handle_status_command(*branch, *compact, group).await,
246            Commands::Foreach {
247                command,
248                args,
249                group,
250                parallel,
251                jobs,
252                continue_on_error,
253            } => {
254                self.handle_foreach_command(
255                    command,
256                    args,
257                    group,
258                    *parallel,
259                    *jobs,
260                    *continue_on_error,
261                )
262                .await
263            }
264            Commands::Audit {
265                group,
266                parallel,
267                jobs,
268                continue_on_vulnerabilities,
269            } => {
270                self.handle_audit_command(group, *parallel, *jobs, *continue_on_vulnerabilities)
271                    .await
272            }
273            Commands::Log {
274                group,
275                oneline,
276                max_count,
277                since,
278                until,
279            } => {
280                self.handle_log_command(group, *oneline, *max_count, since, until)
281                    .await
282            }
283            Commands::DumpManifest {
284                format,
285                output,
286                pretty,
287            } => {
288                self.handle_dump_manifest_command(format, output, *pretty)
289                    .await
290            }
291            Commands::ApplyManifest {
292                manifest_file,
293                force,
294                dry_run,
295            } => {
296                self.handle_apply_manifest_command(manifest_file, *force, *dry_run)
297                    .await
298            }
299        }
300    }

315    async fn handle_sync_command(
316        &self,
317        groups: &[String],
318        force: bool,
319        no_correct_branch: bool,
320        jobs: Option<usize>,
321    ) -> anyhow::Result<()> {
322        // Load workspace
323        let mut workspace = self.load_workspace().await?;
324
325        // Prepare groups list
326        let groups_list = if groups.is_empty() {
327            None
328        } else {
329            Some(groups.to_vec())
330        };
331
332        // Create configuration
333        let config = SyncRepositoriesConfig {
334            groups: groups_list,
335            force,
336            no_correct_branch,
337            parallel_jobs: jobs,
338            verbose: self.cli.verbose,
339        };
340
341        // Execute the use case
342        let use_case = SyncRepositoriesUseCase::new(config);
343
344        println!("{} Synchronizing repositories...", "::".blue().bold());
345
346        match use_case.execute(&mut workspace).await {
347            Ok(result) => {
348                println!("{} Synchronization completed!", "✓".green().bold());
349                if self.cli.verbose {
350                    println!("  Repositories synced: {}", result.synced_count);
351                    println!("  New repositories cloned: {}", result.cloned_count);
352                    println!("  Repositories updated: {}", result.updated_count);
353                    if result.skipped_count > 0 {
354                        println!("  Repositories skipped: {}", result.skipped_count);
355                    }
356                }
357
358                // Show any errors
359                if !result.errors.is_empty() {
360                    println!("{} Some errors occurred:", "⚠".yellow().bold());
361                    for error in result.errors {
362                        println!("  {}", error.red());
363                    }
364                }
365
366                Ok(())
367            }
368            Err(SyncRepositoriesError::WorkspaceNotInitialized(path)) => Err(anyhow::anyhow!(
369                "Workspace not initialized at: {}\nManifest file not found",
370                path
371            )),
372            Err(e) => Err(anyhow::anyhow!("Failed to synchronize repositories: {}", e)),
373        }
374    }

420    async fn handle_foreach_command(
421        &self,
422        command: &str,
423        args: &[String],
424        groups: &[String],
425        parallel: bool,
426        jobs: Option<usize>,
427        continue_on_error: bool,
428    ) -> anyhow::Result<()> {
429        // Load workspace
430        let workspace = self.load_workspace().await?;
431
432        // Prepare groups list
433        let groups_list = if groups.is_empty() {
434            None
435        } else {
436            Some(groups.to_vec())
437        };
438
439        // Build the full command
440        let full_command = if args.is_empty() {
441            command.to_string()
442        } else {
443            format!("{} {}", command, args.join(" "))
444        };
445
446        // Create configuration
447        let config = ForeachCommandConfig {
448            command: full_command,
449            groups: groups_list,
450            parallel,
451            max_parallel: jobs,
452            continue_on_error,
453            verbose: self.cli.verbose,
454            ..Default::default()
455        };
456
457        // Execute the use case
458        let use_case = ForeachCommandUseCase::new(config);
459
460        println!("{} Running command: {}", "::".blue().bold(), command);
461
462        match use_case.execute(&workspace).await {
463            Ok(result) => {
464                println!("{} Command execution completed!", "✓".green().bold());
465                if self.cli.verbose {
466                    println!("  Successful executions: {}", result.success_count);
467                    println!("  Failed executions: {}", result.failure_count);
468                    println!("  Skipped executions: {}", result.skipped_count);
469                }
470
471                // Show any errors
472                let failed_results = result.failed_results();
473                if !failed_results.is_empty() {
474                    println!("{} Some commands failed:", "⚠".yellow().bold());
475                    for result in failed_results {
476                        let default_error = "Unknown error".to_string();
477                        let error_msg = result.error_message.as_ref().unwrap_or(&default_error);
478                        println!("  {}: {}", result.dest.bold(), error_msg.red());
479                    }
480                }
481
482                Ok(())
483            }
484            Err(ForeachCommandError::WorkspaceNotInitialized(path)) => Err(anyhow::anyhow!(
485                "Workspace not initialized at: {}\nManifest file not found",
486                path
487            )),
488            Err(e) => Err(anyhow::anyhow!("Failed to execute command: {}", e)),
489        }
490    }

578    /// Load workspace from the current directory
579    async fn load_workspace(&self) -> anyhow::Result<Workspace> {
580        let current_dir = env::current_dir()?;
581
582        // Try to find manifest.yml in current directory first, then .wmgr/
583        let manifest_file = if current_dir.join("manifest.yml").exists() {
584            current_dir.join("manifest.yml")
585        } else if current_dir.join(".wmgr").join("manifest.yml").exists() {
586            current_dir.join(".wmgr").join("manifest.yml")
587        } else {
588            return Err(anyhow::anyhow!(
589                "Manifest file not found at: {} or {}",
590                current_dir.join("manifest.yml").display(),
591                current_dir.join(".wmgr").join("manifest.yml").display()
592            ));
593        };
594
595        // Load manifest file
596        use crate::domain::entities::workspace::{WorkspaceConfig, WorkspaceStatus};
597        use crate::infrastructure::filesystem::manifest_store::ManifestStore;
598        let mut manifest_store = ManifestStore::new();
599        let processed_manifest = manifest_store
600            .read_manifest(&manifest_file)
601            .await
602            .map_err(|e| anyhow::anyhow!("Failed to load manifest: {}", e))?;
603
604        // Create a simple workspace configuration
605        let workspace_config = WorkspaceConfig::new(&manifest_file.display().to_string(), "main");
606
607        let workspace = Workspace::new(current_dir, workspace_config)
608            .with_status(WorkspaceStatus::Initialized)
609            .with_manifest(processed_manifest.manifest);
610        Ok(workspace)
611    }

613    fn print_compact_status(
614        &self,
615        status: &crate::application::use_cases::status_check::StatusResult,
616    ) {
617        for repo_status in &status.repositories {
618            let state_char = match repo_status.state {
619                crate::application::use_cases::status_check::RepositoryState::Clean => "✓".green(),
620                crate::application::use_cases::status_check::RepositoryState::Dirty => "M".yellow(),
621                crate::application::use_cases::status_check::RepositoryState::Missing => "?".red(),
622                crate::application::use_cases::status_check::RepositoryState::WrongBranch => {
623                    "B".cyan()
624                }
625                crate::application::use_cases::status_check::RepositoryState::OutOfSync => {
626                    "S".magenta()
627                }
628                crate::application::use_cases::status_check::RepositoryState::Error => "E".red(),
629            };
630            println!("{} {}", state_char, repo_status.dest);
631        }
632    }

634    fn print_detailed_status(
635        &self,
636        status: &crate::application::use_cases::status_check::StatusResult,
637        show_branch: bool,
638    ) {
639        for repo_status in &status.repositories {
640            let state_text = match repo_status.state {
641                crate::application::use_cases::status_check::RepositoryState::Clean => {
642                    "clean".green()
643                }
644                crate::application::use_cases::status_check::RepositoryState::Dirty => {
645                    "dirty".yellow()
646                }
647                crate::application::use_cases::status_check::RepositoryState::Missing => {
648                    "missing".red()
649                }
650                crate::application::use_cases::status_check::RepositoryState::WrongBranch => {
651                    "wrong branch".cyan()
652                }
653                crate::application::use_cases::status_check::RepositoryState::OutOfSync => {
654                    "out of sync".magenta()
655                }
656                crate::application::use_cases::status_check::RepositoryState::Error => {
657                    "error".red()
658                }
659            };
660
661            print!("{}: {}", repo_status.dest.bold(), state_text);
662
663            if show_branch {
664                if let Some(ref current_branch) = repo_status.current_branch {
665                    print!(" ({})", current_branch.blue());
666                }
667            }
668
669            if repo_status.state
670                == crate::application::use_cases::status_check::RepositoryState::Dirty
671            {
672                let mut changes = Vec::new();
673                if repo_status.modified_files > 0 {
674                    changes.push(format!("{}M", repo_status.modified_files));
675                }
676                if repo_status.staged_files > 0 {
677                    changes.push(format!("{}S", repo_status.staged_files));
678                }
679                if repo_status.untracked_files > 0 {
680                    changes.push(format!("{}U", repo_status.untracked_files));
681                }
682                if !changes.is_empty() {
683                    print!(" [{}]", changes.join(" "));
684                }
685            }
686
687            println!();
688        }
689    }
