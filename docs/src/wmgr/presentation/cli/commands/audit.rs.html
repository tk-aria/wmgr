50    /// Execute the audit command
51    pub async fn execute(&self, args: AuditArgs) -> Result<(), Box<dyn std::error::Error>> {
52        // 1. Determine workspace directory
53        let workspace_dir = args
54            .workspace_dir
55            .unwrap_or_else(|| std::env::current_dir().unwrap());
56
57        // 2. Load workspace configuration
58        let workspace = self.load_workspace(&workspace_dir).await?;
59
60        // 3. Create audit configuration
61        let config = SecurityAuditConfig::new()
62            .with_groups(args.groups.unwrap_or_default())
63            .with_parallel(args.parallel, args.max_parallel)
64            .with_fail_on_vulnerabilities(!args.continue_on_vulnerabilities)
65            .with_verbose(args.verbose);
66
67        // 4. Execute audit
68        let use_case = SecurityAuditUseCase::new(config);
69
70        if args.verbose {
71            println!("Starting security audit...");
72        }
73
74        match use_case.execute(&workspace).await {
75            Ok(result) => {
76                self.display_audit_result(&result, args.verbose);
77
78                if result.has_critical_or_high_vulnerabilities()
79                    && !args.continue_on_vulnerabilities
80                {
81                    std::process::exit(1);
82                }
83
84                Ok(())
85            }
86            Err(SecurityAuditError::NoRustProjectsFound) => {
87                self.display.warning("No Rust projects found in workspace");
88                Ok(())
89            }
90            Err(e) => {
91                eprintln!("Security audit failed: {}", e);
92                Err(Box::new(e))
93            }
94        }
95    }

97    /// Load workspace from directory
98    async fn load_workspace(
99        &self,
100        workspace_dir: &PathBuf,
101    ) -> Result<Workspace, Box<dyn std::error::Error>> {
102        let workspace = Workspace::new(workspace_dir.clone(), WorkspaceConfig::default_local());
103
104        // Use workspace.manifest_file_path() to support wmgr.yml, wmgr.yaml, manifest.yml, manifest.yaml
105        let manifest_file = workspace.manifest_file_path();
106
107        if !manifest_file.exists() {
108            return Err("Manifest file not found. Tried wmgr.yml, wmgr.yaml, manifest.yml, manifest.yaml in current directory and .wmgr/ subdirectory".into());
109        }
110
111        // Load manifest file
112        use crate::domain::entities::workspace::{WorkspaceConfig, WorkspaceStatus};
113        use crate::infrastructure::filesystem::manifest_store::ManifestStore;
114        let mut manifest_store = ManifestStore::new();
115
116        let processed_manifest = manifest_store
117            .read_manifest(&manifest_file)
118            .await
119            .map_err(|e| format!("Failed to read manifest: {}", e))?;
120
121        // Create workspace config from manifest
122        let workspace_config = WorkspaceConfig::new(
123            "file://".to_string() + &manifest_file.to_string_lossy(),
124            processed_manifest
125                .manifest
126                .default_branch
127                .clone()
128                .unwrap_or_else(|| "main".to_string()),
129        );
130
131        let workspace = Workspace::new(workspace_dir.clone(), workspace_config)
132            .with_status(WorkspaceStatus::Initialized)
133            .with_manifest(processed_manifest.manifest);
134
135        Ok(workspace)
136    }

138    /// Display audit results
139    fn display_audit_result(
140        &self,
141        result: &crate::application::use_cases::security_audit::WorkspaceAuditResult,
142        verbose: bool,
143    ) {
144        // Summary
145        println!("\n=== Security Audit Summary ===");
146        println!("Total repositories: {}", result.total_count());
147        println!("Audited (Rust projects): {}", result.audited_count);
148        println!("Skipped (non-Rust): {}", result.skipped_count);
149        println!("Errors: {}", result.error_count);
150        println!("With vulnerabilities: {}", result.vulnerable_count);
151
152        // Vulnerability details
153        if result.has_vulnerabilities() {
154            println!("\n=== Vulnerabilities Found ===");
155
156            let vulnerable_repos = result.vulnerable_results();
157            for repo_result in &vulnerable_repos {
158                if let Some(audit_result) = &repo_result.audit_result {
159                    let count = &audit_result.warning_count;
160
161                    // Repository header
162                    if count.has_critical_or_high() {
163                        eprintln!("📛 {}: {} vulnerabilities", repo_result.dest, count.total());
164                    } else {
165                        self.display.warning(&format!(
166                            "⚠️  {}: {} vulnerabilities",
167                            repo_result.dest,
168                            count.total()
169                        ));
170                    }
171
172                    // Severity breakdown
173                    if count.critical > 0 {
174                        println!("    🔴 Critical: {}", count.critical);
175                    }
176                    if count.high > 0 {
177                        println!("    🟠 High: {}", count.high);
178                    }
179                    if count.medium > 0 {
180                        println!("    🟡 Medium: {}", count.medium);
181                    }
182                    if count.low > 0 {
183                        println!("    🔵 Low: {}", count.low);
184                    }
185
186                    // Detailed vulnerability information if verbose
187                    if verbose && !audit_result.vulnerabilities.is_empty() {
188                        println!("    Vulnerabilities:");
189                        for vuln in &audit_result.vulnerabilities {
190                            println!(
191                                "      - {} ({}): {}",
192                                vuln.id,
193                                format!("{:?}", vuln.severity).to_lowercase(),
194                                vuln.description.chars().take(80).collect::<String>()
195                            );
196                            if let Some(url) = &vuln.url {
197                                println!("        More info: {}", url);
198                            }
199                        }
200                    }
201                }
202            }
203
204            // Recommendations
205            println!("\n=== Recommendations ===");
206            if result.has_critical_or_high_vulnerabilities() {
207                eprintln!("❗ Critical or high severity vulnerabilities found!");
208                println!("   • Review and update affected dependencies immediately");
209                println!("   • Run 'cargo update' in affected repositories");
210                println!("   • Check for security advisories and patches");
211            } else {
212                self.display
213                    .warning("⚠️  Lower severity vulnerabilities found");
214                println!("   • Consider updating dependencies when convenient");
215                println!("   • Monitor for security updates");
216            }
217
218            println!("   • Run 'cargo audit' in individual repositories for more details");
219            println!("   • Use 'cargo audit fix' to attempt automatic fixes");
220        } else if result.audited_count > 0 {
221            self.display.success("✅ No vulnerabilities found!");
222        }
223
224        // Error details
225        if result.error_count > 0 {
226            println!("\n=== Audit Errors ===");
227            let failed_repos = result.failed_results();
228            for repo_result in failed_repos {
229                if let Some(error) = &repo_result.error {
230                    eprintln!("❌ {}: {}", repo_result.dest, error);
231                }
232            }
233        }
234
235        // Final status
236        println!();
237        if result.has_critical_or_high_vulnerabilities() {
238            eprintln!("🚨 Security audit completed with critical/high vulnerabilities");
239        } else if result.has_vulnerabilities() {
240            self.display
241                .warning("⚠️  Security audit completed with minor vulnerabilities");
242        } else if result.is_success() && result.audited_count > 0 {
243            self.display
244                .success("✅ Security audit completed successfully - no vulnerabilities found");
245        } else {
246            println!("ℹ️  Security audit completed");
247        }
248    }
