19impl SyncCommand {
20    pub fn new(
21        groups: Vec<String>,
22        force: bool,
23        no_correct_branch: bool,
24        jobs: Option<usize>,
25        verbose: bool,
26    ) -> Self {

36    pub async fn execute(&self) -> Result<()> {
37        // Load workspace
38        let mut workspace = self.load_workspace().await?;
39
40        // Prepare groups list
41        let groups_list = if self.groups.is_empty() {
42            None
43        } else {
44            Some(self.groups.clone())
45        };
46
47        // Create configuration
48        let config = SyncRepositoriesConfig {
49            groups: groups_list,
50            force: self.force,
51            no_correct_branch: self.no_correct_branch,
52            parallel_jobs: self.jobs,
53            verbose: self.verbose,
54        };
55
56        // Execute the use case
57        let use_case = SyncRepositoriesUseCase::new(config);
58
59        println!("{} Synchronizing repositories...", "::".blue().bold());
60
61        match use_case.execute(&mut workspace).await {
62            Ok(result) => {
63                println!("{} Synchronization completed!", "✓".green().bold());
64                if self.verbose {
65                    println!("  Repositories synced: {}", result.synced_count);
66                    println!("  New repositories cloned: {}", result.cloned_count);
67                    println!("  Repositories updated: {}", result.updated_count);
68                    if result.skipped_count > 0 {
69                        println!("  Repositories skipped: {}", result.skipped_count);
70                    }
71                }
72
73                // Show any errors
74                if !result.errors.is_empty() {
75                    println!("{} Some errors occurred:", "⚠".yellow().bold());
76                    for error in result.errors {
77                        println!("  {}", error.red());
78                    }
79                }
80
81                Ok(())
82            }
83            Err(SyncRepositoriesError::WorkspaceNotInitialized(path)) => Err(anyhow::anyhow!(
84                "Workspace not initialized at: {}\nManifest file not found",
85                path
86            )),
87            Err(e) => Err(anyhow::anyhow!("Failed to synchronize repositories: {}", e)),
88        }
89    }
90
91    /// Load workspace from the current directory
92    async fn load_workspace(&self) -> Result<Workspace> {
93        let current_dir = env::current_dir()?;
94
95        let workspace = Workspace::new(current_dir.clone(), WorkspaceConfig::default_local());
96
97        // Use workspace.manifest_file_path() to support wmgr.yml, wmgr.yaml, manifest.yml, manifest.yaml
98        let manifest_file = workspace.manifest_file_path();
99
100        if !manifest_file.exists() {
101            return Err(anyhow::anyhow!("Manifest file not found. Tried wmgr.yml, wmgr.yaml, manifest.yml, manifest.yaml in current directory and .wmgr/ subdirectory"));
102        }
103
104        // Load manifest file
105        use crate::domain::entities::workspace::{WorkspaceConfig, WorkspaceStatus};
106        use crate::infrastructure::filesystem::manifest_store::ManifestStore;
107        let mut manifest_store = ManifestStore::new();
108        let processed_manifest = manifest_store
109            .read_manifest(&manifest_file)
110            .await
111            .map_err(|e| anyhow::anyhow!("Failed to load manifest: {}", e))?;
112
113        // Create a simple workspace configuration
114        let workspace_config = WorkspaceConfig::new(&manifest_file.display().to_string(), "main");
115
116        let workspace = Workspace::new(current_dir, workspace_config)
117            .with_status(WorkspaceStatus::Initialized)
118            .with_manifest(processed_manifest.manifest);
119        Ok(workspace)
120    }
