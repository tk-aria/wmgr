36    pub async fn execute(&self) -> Result<()> {
37        // Load workspace
38        let workspace = self.load_workspace().await?;
39
40        // Get repositories to check
41        let repositories = self.get_repositories_to_check(&workspace)?;
42
43        if repositories.is_empty() {
44            println!("{} No repositories found to check log", "⚠".yellow().bold());
45            return Ok(());
46        }
47
48        println!(
49            "{} Showing commit log for {} repositories",
50            "::".blue().bold(),
51            repositories.len()
52        );
53
54        // Show log for each repository
55        for repo in repositories {
56            self.show_repository_log(&repo, &workspace).await?;
57        }
58
59        Ok(())
60    }

62    async fn show_repository_log(
63        &self,
64        repo: &crate::domain::entities::repository::Repository,
65        workspace: &Workspace,
66    ) -> Result<()> {
67        let repo_path = workspace.root_path.join(&repo.dest);
68
69        if !repo_path.exists() {
70            if self.verbose {
71                println!("{} {}: repository not found", "⚠".yellow(), repo.dest);
72            }
73            return Ok(());
74        }
75
76        println!();
77        println!("{} {}", "Repository:".bold(), repo.dest.green());
78
79        // Try to open the git repository
80        let git_repo = match git2::Repository::open(&repo_path) {
81            Ok(repo) => repo,
82            Err(e) => {
83                if self.verbose {
84                    println!("  {}: not a git repository - {}", "Error".red(), e);
85                }
86                return Ok(());
87            }
88        };
89
90        // Get the current branch
91        let head_ref = match git_repo.head() {
92            Ok(head) => head,
93            Err(e) => {
94                if self.verbose {
95                    println!("  {}: failed to get HEAD - {}", "Error".red(), e);
96                }
97                return Ok(());
98            }
99        };
100
101        let branch_name = head_ref.shorthand().unwrap_or("unknown");
102        if self.verbose {
103            println!("  {}: {}", "Branch".blue(), branch_name);
104        }
105
106        // Walk commits
107        let mut revwalk = git_repo.revwalk()?;
108        revwalk.push_head()?;
109        revwalk.set_sorting(git2::Sort::TIME)?;
110
111        let mut count = 0;
112        let max_count = self.max_count.unwrap_or(10);
113
114        for commit_oid in revwalk {
115            if count >= max_count {
116                break;
117            }
118
119            let commit_oid = commit_oid?;
120            let commit = git_repo.find_commit(commit_oid)?;
121
122            // Check date filters
123            let commit_time = commit.time();
124            let commit_timestamp = commit_time.seconds();
125
126            if let Some(ref since) = self.since {
127                // For simplicity, we'll just show a placeholder for date parsing
128                // In a real implementation, you'd parse the date string
129                if self.verbose {
130                    println!(
131                        "  (Date filtering with 'since: {}' not fully implemented)",
132                        since
133                    );
134                }
135            }
136
137            if let Some(ref until) = self.until {
138                // For simplicity, we'll just show a placeholder for date parsing
139                // In a real implementation, you'd parse the date string
140                if self.verbose {
141                    println!(
142                        "  (Date filtering with 'until: {}' not fully implemented)",
143                        until
144                    );
145                }
146            }
147
148            // Display commit info
149            let commit_hash = commit.id().to_string();
150            let short_hash = &commit_hash[..7];
151            let message = commit.message().unwrap_or("(no message)");
152            let summary = message.lines().next().unwrap_or("(no message)");
153
154            if self.oneline {
155                println!("  {} {}", short_hash.yellow(), summary);
156            } else {
157                let author = commit.author();
158                let author_name = author.name().unwrap_or("unknown");
159                let author_email = author.email().unwrap_or("unknown");
160
161                // Format time
162                let datetime = chrono::DateTime::from_timestamp(commit_timestamp, 0)
163                    .unwrap_or_else(|| chrono::DateTime::from_timestamp(0, 0).unwrap());
164                let formatted_time = datetime.format("%Y-%m-%d %H:%M:%S");
165
166                println!("  {} {}", "commit".yellow(), commit_hash.yellow());
167                println!("  {}: {} <{}>", "Author".blue(), author_name, author_email);
168                println!("  {}: {}", "Date".blue(), formatted_time);
169                println!();
170
171                // Show full commit message with indentation
172                for line in message.lines() {
173                    println!("      {}", line);
174                }
175                println!();
176            }
177
178            count += 1;
179        }
180
181        if count == 0 {
182            println!("  {}", "No commits found".dimmed());
183        }
184
185        Ok(())
186    }

209    /// Load workspace from the current directory
210    async fn load_workspace(&self) -> Result<Workspace> {
211        let current_dir = env::current_dir()?;
212
213        let workspace = Workspace::new(current_dir.clone(), WorkspaceConfig::default_local());
214
215        // Use workspace.manifest_file_path() to support wmgr.yml, wmgr.yaml, manifest.yml, manifest.yaml
216        let manifest_file = workspace.manifest_file_path();
217
218        if !manifest_file.exists() {
219            return Err(anyhow::anyhow!("Manifest file not found. Tried wmgr.yml, wmgr.yaml, manifest.yml, manifest.yaml in current directory and .wmgr/ subdirectory"));
220        }
221
222        // Load manifest file
223        use crate::domain::entities::workspace::{WorkspaceConfig, WorkspaceStatus};
224        use crate::infrastructure::filesystem::manifest_store::ManifestStore;
225        let mut manifest_store = ManifestStore::new();
226
227        let processed_manifest = manifest_store
228            .read_manifest(&manifest_file)
229            .await
230            .map_err(|e| anyhow::anyhow!("Failed to read manifest: {}", e))?;
231
232        // Create workspace config from manifest
233        let workspace_config = WorkspaceConfig::new(
234            "file://".to_string() + &manifest_file.to_string_lossy(),
235            processed_manifest
236                .manifest
237                .default_branch
238                .clone()
239                .unwrap_or_else(|| "main".to_string()),
240        );
241
242        let workspace = Workspace::new(current_dir, workspace_config)
243            .with_status(WorkspaceStatus::Initialized)
244            .with_manifest(processed_manifest.manifest);
245
246        Ok(workspace)
247    }
