18impl StatusCommand {
19    pub fn new(groups: Vec<String>, show_branch: bool, compact: bool, verbose: bool) -> Self {
20        Self {
21            groups,
22            show_branch,
23            compact,
24            verbose,
25        }

28    pub async fn execute(&self) -> Result<()> {
29        // Load workspace
30        let workspace = self.load_workspace().await?;
31
32        // Prepare groups list
33        let groups_list = if self.groups.is_empty() {
34            None
35        } else {
36            Some(self.groups.clone())
37        };
38
39        // Create configuration
40        let config = StatusCheckConfig {
41            groups: groups_list,
42            show_branch: self.show_branch,
43            compact: self.compact,
44            verbose: self.verbose,
45        };
46
47        // Execute the use case
48        let use_case = StatusCheckUseCase::new(config);
49
50        match use_case.execute(&workspace).await {
51            Ok(status) => {
52                if self.compact {
53                    self.print_compact_status(&status);
54                } else {
55                    self.print_detailed_status(&status);
56                }
57                Ok(())
58            }
59            Err(StatusCheckError::WorkspaceNotInitialized(path)) => Err(anyhow::anyhow!(
60                "Workspace not initialized at: {}\nManifest file not found",
61                path
62            )),
63            Err(e) => Err(anyhow::anyhow!("Failed to check status: {}", e)),
64        }
65    }
66
67    fn print_compact_status(
68        &self,
69        status: &crate::application::use_cases::status_check::StatusResult,
70    ) {
71        for repo_status in &status.repositories {
72            let state_char = match repo_status.state {
73                RepositoryState::Clean => "âœ“".green(),
74                RepositoryState::Dirty => "M".yellow(),
75                RepositoryState::Missing => "?".red(),
76                RepositoryState::WrongBranch => "B".cyan(),
77                RepositoryState::OutOfSync => "S".magenta(),
78                RepositoryState::Error => "E".red(),
79            };
80            println!("{} {}", state_char, repo_status.dest);
81        }
82    }
83
84    fn print_detailed_status(
85        &self,
86        status: &crate::application::use_cases::status_check::StatusResult,
87    ) {
88        for repo_status in &status.repositories {
89            let state_text = match repo_status.state {
90                RepositoryState::Clean => "clean".green(),
91                RepositoryState::Dirty => "dirty".yellow(),
92                RepositoryState::Missing => "missing".red(),
93                RepositoryState::WrongBranch => "wrong branch".cyan(),
94                RepositoryState::OutOfSync => "out of sync".magenta(),
95                RepositoryState::Error => "error".red(),
96            };
97
98            print!("{}: {}", repo_status.dest.bold(), state_text);
99
100            if self.show_branch {
101                if let Some(ref current_branch) = repo_status.current_branch {
102                    print!(" ({})", current_branch.blue());
103                }
104            }
105
106            if repo_status.state == RepositoryState::Dirty {
107                let mut changes = Vec::new();
108                if repo_status.modified_files > 0 {
109                    changes.push(format!("{}M", repo_status.modified_files));
110                }
111                if repo_status.staged_files > 0 {
112                    changes.push(format!("{}S", repo_status.staged_files));
113                }
114                if repo_status.untracked_files > 0 {
115                    changes.push(format!("{}U", repo_status.untracked_files));
116                }
117                if !changes.is_empty() {
118                    print!(" [{}]", changes.join(" "));
119                }
120            }
121
122            println!();
123        }
124    }

126    /// Load workspace from the current directory
127    async fn load_workspace(&self) -> Result<Workspace> {
128        let current_dir = env::current_dir()?;
129
130        let workspace = Workspace::new(current_dir.clone(), WorkspaceConfig::default_local());
131
132        // Use workspace.manifest_file_path() to support wmgr.yml, wmgr.yaml, manifest.yml, manifest.yaml
133        let manifest_file = workspace.manifest_file_path();
134
135        if !manifest_file.exists() {
136            return Err(anyhow::anyhow!("Manifest file not found. Tried wmgr.yml, wmgr.yaml, manifest.yml, manifest.yaml in current directory and .wmgr/ subdirectory"));
137        }
138
139        // Load manifest file
140        use crate::domain::entities::workspace::{WorkspaceConfig, WorkspaceStatus};
141        use crate::infrastructure::filesystem::manifest_store::ManifestStore;
142        let mut manifest_store = ManifestStore::new();
143        let processed_manifest = manifest_store
144            .read_manifest(&manifest_file)
145            .await
146            .map_err(|e| anyhow::anyhow!("Failed to load manifest: {}", e))?;
147
148        // Create a simple workspace configuration
149        let workspace_config = WorkspaceConfig::new(&manifest_file.display().to_string(), "main");
150
151        let workspace = Workspace::new(current_dir, workspace_config)
152            .with_status(WorkspaceStatus::Initialized)
153            .with_manifest(processed_manifest.manifest);
154        Ok(workspace)
155    }
