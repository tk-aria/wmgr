225    /// 文字列からマニフェストを解析
226    pub async fn parse_from_string(
227        &mut self,
228        content: &str,
229        base_path: Option<PathBuf>,
230    ) -> Result<ProcessedManifest, ManifestServiceError> {
231        // 基本のYAMLパース
232        let extended_manifest: ExtendedManifest = serde_yaml::from_str(content)?;
233
234        // バリデーション
235        self.validate_manifest(&extended_manifest.manifest)?;
236
237        // Deep manifest処理
238        let processed = self
239            .process_deep_manifest(extended_manifest, base_path, 0, &mut Vec::new())
240            .await?;
241
242        Ok(processed)
243    }

245    /// マニフェストを検証
246    pub fn validate_manifest(&self, manifest: &Manifest) -> Result<(), ManifestServiceError> {
247        // 重複するdestの検証
248        let mut dest_set = std::collections::HashSet::new();
249        for repo in &manifest.repos {
250            if !dest_set.insert(&repo.dest) {
251                return Err(ManifestServiceError::ValidationError(format!(
252                    "Duplicate destination path: {}",
253                    repo.dest
254                )));
255            }
256
257            // URL検証
258            GitUrl::new(&repo.url)?;
259        }
260
261        // グループ検証
262        if let Some(groups) = &manifest.groups {
263            for (group_name, group) in groups {
264                for repo_dest in &group.repos {
265                    if !manifest.repos.iter().any(|r| &r.dest == repo_dest) {
266                        return Err(ManifestServiceError::ValidationError(format!(
267                            "Group '{}' references non-existent repository: {}",
268                            group_name, repo_dest
269                        )));
270                    }
271                }
272            }
273        }
274
275        Ok(())
276    }

278    /// 指定されたグループでマニフェストをフィルタリング
279    pub fn filter_by_groups(
280        &self,
281        manifest: &Manifest,
282        group_names: &[String],
283    ) -> Result<Manifest, ManifestServiceError> {
284        if group_names.is_empty() {
285            return Ok(manifest.clone());
286        }
287
288        let mut filtered_repos = Vec::new();
289        let mut repo_dests = std::collections::HashSet::new();
290
291        // 各グループからリポジトリを収集
292        for group_name in group_names {
293            let group_repos = manifest.get_repos_in_group(group_name);
294            if group_repos.is_empty() {
295                return Err(ManifestServiceError::GroupNotFound(group_name.clone()));
296            }
297
298            for repo in group_repos {
299                if repo_dests.insert(&repo.dest) {
300                    filtered_repos.push(repo.clone());
301                }
302            }
303        }
304
305        // フィルタされたグループ定義も作成
306        let mut filtered_groups = HashMap::new();
307        if let Some(groups) = &manifest.groups {
308            for group_name in group_names {
309                if let Some(group) = groups.get(group_name) {
310                    // グループ内のリポジトリをフィルタされたもののみに限定
311                    let filtered_group_repos: Vec<String> = group
312                        .repos
313                        .iter()
314                        .filter(|dest| repo_dests.contains(dest))
315                        .cloned()
316                        .collect();
317
318                    if !filtered_group_repos.is_empty() {
319                        let mut filtered_group = group.clone();
320                        filtered_group.repos = filtered_group_repos;
321                        filtered_groups.insert(group_name.clone(), filtered_group);
322                    }
323                }
324            }
325        }
326
327        let mut filtered_manifest = Manifest::new(filtered_repos);
328        filtered_manifest.default_branch = manifest.default_branch.clone();
329
330        if !filtered_groups.is_empty() {
331            filtered_manifest = filtered_manifest.with_groups(filtered_groups);
332        }
333
334        Ok(filtered_manifest)
335    }

361    /// Deep manifestを処理
362    fn process_deep_manifest<'a>(
363        &'a mut self,
364        mut extended_manifest: ExtendedManifest,
365        base_path: Option<PathBuf>,
366        depth: usize,
367        visited: &'a mut Vec<String>,
368    ) -> std::pin::Pin<
369        Box<dyn std::future::Future<Output = Result<ProcessedManifest, ManifestServiceError>> + 'a>,
370    > {
371        Box::pin(async move {
372            // 深度制限チェック
373            if depth > self.options.max_depth {
374                return Err(ManifestServiceError::DepthLimitExceeded {
375                    max: self.options.max_depth,
376                    current: depth,
377                });
378            }
379
380            let mut warnings = Vec::new();
381            let mut all_includes = Vec::new();
382
383            // Future manifest設定の処理
384            if let Some(future_config) = &extended_manifest.future {
385                warnings.extend(self.process_future_config(future_config));
386            }
387
388            // インクルードの処理
389            if let Some(includes) = &extended_manifest.includes {
390                for include in includes {
391                    let include_url = self.resolve_include_url(&include.url, &base_path)?;
392
393                    // 循環依存チェック
394                    if self.options.detect_circular_dependencies && visited.contains(&include_url) {
395                        return Err(ManifestServiceError::CircularDependency(format!(
396                            "Circular dependency detected: {} -> {}",
397                            visited.join(" -> "),
398                            include_url
399                        )));
400                    }
401
402                    visited.push(include_url.clone());
403
404                    // インクルードマニフェストを取得・処理
405                    let included_content = self.fetch_remote_content(&include_url).await?;
406                    let included_extended: ExtendedManifest =
407                        serde_yaml::from_str(&included_content)?;
408
409                    let included_processed = self
410                        .process_deep_manifest(included_extended, None, depth + 1, visited)
411                        .await?;
412
413                    visited.pop();
414
415                    // インクルードされたマニフェストをマージ
416                    let filtered_manifest = if let Some(groups) = &include.groups {
417                        self.filter_by_groups(&included_processed.manifest, groups)?
418                    } else {
419                        included_processed.manifest
420                    };
421
422                    // インクルード情報を記録（マージ前に）
423                    let repo_count = filtered_manifest.repos.len();
424
425                    extended_manifest.manifest = self.merge_manifests(
426                        extended_manifest.manifest,
427                        filtered_manifest,
428                        include.priority,
429                    )?;
430                    all_includes.push(IncludeInfo {
431                        url: include_url,
432                        revision: include
433                            .revision
434                            .clone()
435                            .unwrap_or_else(|| "HEAD".to_string()),
436                        repo_count,
437                    });
438
439                    warnings.extend(included_processed.warnings);
440                    all_includes.extend(included_processed.includes);
441                }
442            }
443
444            Ok(ProcessedManifest {
445                manifest: extended_manifest.manifest,
446                warnings,
447                includes: all_includes,
448            })
449        })
450    }

452    /// Future manifest設定を処理
453    fn process_future_config(&self, future_config: &FutureManifestConfig) -> Vec<String> {
454        let mut warnings = Vec::new();
455
456        // バージョンチェック（簡易実装）
457        if let Some(min_version) = &future_config.min_version {
458            warnings.push(format!("Minimum version requirement: {}", min_version));
459        }
460
461        // 廃止予定警告
462        if let Some(deprecated) = &future_config.deprecated {
463            for dep in deprecated {
464                let warning = if let Some(removal_version) = &dep.removal_version {
465                    format!(
466                        "DEPRECATED: {} (will be removed in {}): {}",
467                        dep.feature, removal_version, dep.message
468                    )
469                } else {
470                    format!("DEPRECATED: {}: {}", dep.feature, dep.message)
471                };
472                warnings.push(warning);
473            }
474        }
475
476        warnings
477    }

479    /// インクルードURLを解決
480    fn resolve_include_url(
481        &self,
482        url: &str,
483        base_path: &Option<PathBuf>,
484    ) -> Result<String, ManifestServiceError> {
485        // 既に絶対URLの場合はそのまま返す
486        if url.starts_with("http://") || url.starts_with("https://") || url.starts_with("git@") {
487            return Ok(url.to_string());
488        }
489
490        // 相対パスの場合、base_pathからの相対パスとして解決
491        if let Some(base) = base_path {
492            if let Some(parent) = base.parent() {
493                let resolved = parent.join(url);
494                return Ok(resolved.to_string_lossy().to_string());
495            }
496        }
497
498        Ok(url.to_string())
499    }

501    /// リモートコンテンツを取得
502    async fn fetch_remote_content(&self, url: &str) -> Result<String, ManifestServiceError> {
503        if !self.options.enable_remote_fetch {
504            return Err(ManifestServiceError::RemoteManifestFetchFailed {
505                url: url.to_string(),
506                reason: "Remote fetch is disabled".to_string(),
507            });
508        }
509
510        if url.starts_with("http://") || url.starts_with("https://") {
511            let response = self.http_client.get(url).send().await?;
512
513            if response.status().is_success() {
514                Ok(response.text().await?)
515            } else {
516                Err(ManifestServiceError::RemoteManifestFetchFailed {
517                    url: url.to_string(),
518                    reason: format!("HTTP {}", response.status()),
519                })
520            }
521        } else {
522            // ローカルファイルとして扱う
523            match tokio::fs::read_to_string(url).await {
524                Ok(content) => Ok(content),
525                Err(e) => Err(ManifestServiceError::RemoteManifestFetchFailed {
526                    url: url.to_string(),
527                    reason: e.to_string(),
528                }),
529            }
530        }
531    }

533    /// マニフェストをマージ
534    fn merge_manifests(
535        &self,
536        mut base: Manifest,
537        include: Manifest,
538        _priority: i32,
539    ) -> Result<Manifest, ManifestServiceError> {
540        // リポジトリをマージ（重複するdestは基本的にbaseを優先）
541        let mut dest_set: std::collections::HashSet<String> =
542            base.repos.iter().map(|r| r.dest.clone()).collect();
543
544        for repo in include.repos {
545            if !dest_set.contains(&repo.dest) {
546                dest_set.insert(repo.dest.clone());
547                base.repos.push(repo);
548            }
549        }
550
551        // グループをマージ
552        if let Some(include_groups) = include.groups {
553            let mut merged_groups = base.groups.unwrap_or_default();
554
555            for (group_name, group) in include_groups {
556                // 既存のグループがある場合はリポジトリをマージ
557                if let Some(existing_group) = merged_groups.get_mut(&group_name) {
558                    for repo_dest in group.repos {
559                        if !existing_group.repos.contains(&repo_dest) {
560                            existing_group.repos.push(repo_dest);
561                        }
562                    }
563                } else {
564                    merged_groups.insert(group_name, group);
565                }
566            }
567
568            base.groups = Some(merged_groups);
569        }
570
571        // デフォルトブランチはbaseを優先
572        if base.default_branch.is_none() {
573            base.default_branch = include.default_branch;
574        }
575
576        Ok(base)
577    }
