7    #[test]
8    fn test_dot_pattern_detection() {
9        // テストケース1: 単体の "."
10        println!("Testing single dot '.'");
11        let result = FilePath::new(".");
12        match result {
13            Ok(fp) => println!("✓ Success: {}", fp.as_str()),
14            Err(e) => println!("✗ Error: {}", e),
15        }
16
17        // テストケース2: "./"を含むパス
18        println!("Testing path with './'");
19        let result = FilePath::new("./somedir");
20        match result {
21            Ok(fp) => println!("✓ Success: {}", fp.as_str()),
22            Err(e) => println!("✗ Error: {}", e),
23        }
24
25        // テストケース3: ファイル名に.を含むパス（許可されるべき）
26        println!("Testing file with extension");
27        let result = FilePath::new("file.txt");
28        match result {
29            Ok(fp) => println!("✓ Success: {}", fp.as_str()),
30            Err(e) => println!("✗ Error: {}", e),
31        }
32
33        // テストケース4: 正規化後に"."になるパターン
34        println!("Testing normalized dot pattern");
35        let current_dir = std::env::current_dir().unwrap();
36        let repo_path = current_dir.join(".");
37        let path_str = repo_path.to_string_lossy();
38        println!("Full path: {}", path_str);
39        let result = FilePath::new(&path_str);
40        match result {
41            Ok(fp) => println!("✓ Success: {}", fp.as_str()),
42            Err(e) => println!("✗ Error: {}", e),
43        }
44    }

46    #[test]
47    fn test_dangerous_patterns_directly() {
48        let dangerous_patterns = [
49            "|",
50            "&",
51            ";",
52            "$",
53            "`",
54            "$(",
55            "${",
56            "<script",
57            "javascript:",
58            "data:",
59            "%2e%2e",
60            "%2e%2e%2f",
61            "%2e%2e%5c",
62            "..%2f",
63            "..%5c",
64            "%252e%252e",
65            "\u{2024}",
66            "\u{ff0e}",
67            "%252e",
68            "%255c",
69            "%252f",
70        ];
71
72        for pattern in &dangerous_patterns {
73            let test_path = format!("file{}", pattern);
74            let result = FilePath::new(&test_path);
75            println!(
76                "Testing pattern '{}' in path '{}': {:?}",
77                pattern, test_path, result
78            );
79            if let Err(FilePathError::DangerousPattern(detected)) = result {
80                println!("Detected dangerous pattern: {}", detected);
81            }
82        }
83    }
