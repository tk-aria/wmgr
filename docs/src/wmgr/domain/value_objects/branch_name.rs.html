6/// BranchName関連のエラー
7#[derive(Debug, Error, PartialEq)]
8pub enum BranchNameError {
9    #[error("Branch name cannot be empty")]
10    Empty,
11
12    #[error("Branch name too long: {0} characters (max: 255)")]
13    TooLong(usize),
14
15    #[error("Invalid character in branch name: {0}")]
16    InvalidCharacter(String),
17
18    #[error("Branch name cannot start with '-': {0}")]
19    StartsWithHyphen(String),
20
21    #[error("Branch name cannot end with '.lock': {0}")]
22    EndsWithLock(String),
23
24    #[error("Branch name contains consecutive dots: {0}")]
25    ConsecutiveDots(String),
26
27    #[error("Reserved branch name: {0}")]
28    Reserved(String),
29}

31/// Gitブランチ名の値オブジェクト
32#[derive(Debug, Clone, PartialEq, Eq, Hash, Serialize, Deserialize)]
33pub struct BranchName {
34    /// 検証済みブランチ名
35    name: String,
36}
37
38impl BranchName {
39    /// 新しいBranchNameインスタンスを作成
40    pub fn new(name: &str) -> Result<Self, BranchNameError> {
41        Self::validate(name)?;
42        Ok(Self {
43            name: name.to_string(),
44        })
45    }
46
47    /// ブランチ名の妥当性を検証
48    fn validate(name: &str) -> Result<(), BranchNameError> {
49        // 空文字チェック
50        if name.is_empty() {
51            return Err(BranchNameError::Empty);
52        }
53
54        // 長さチェック（255文字まで）
55        if name.len() > 255 {
56            return Err(BranchNameError::TooLong(name.len()));
57        }
58
59        // ハイフンで始まるかチェック
60        if name.starts_with('-') {
61            return Err(BranchNameError::StartsWithHyphen(name.to_string()));
62        }
63
64        // .lockで終わるかチェック
65        if name.ends_with(".lock") {
66            return Err(BranchNameError::EndsWithLock(name.to_string()));
67        }
68
69        // 予約語チェック
70        if matches!(name, "HEAD" | "ORIG_HEAD" | "FETCH_HEAD" | "MERGE_HEAD") {
71            return Err(BranchNameError::Reserved(name.to_string()));
72        }
73
74        // 不正な文字をチェック
75        // ASCII制御文字、スペース、~、^、:、?、*、[、\、DEL
76        for ch in name.chars() {
77            if ch.is_ascii_control()
78                || ch == ' '
79                || ch == '~'
80                || ch == '^'
81                || ch == ':'
82                || ch == '?'
83                || ch == '*'
84                || ch == '['
85                || ch == '\\'
86                || ch == '\x7F'
87            {
88                return Err(BranchNameError::InvalidCharacter(ch.to_string()));
89            }
90        }
91
92        // 連続するドットをチェック
93        if name.contains("..") {
94            return Err(BranchNameError::ConsecutiveDots(name.to_string()));
95        }
96
97        // リモート追跡ブランチの形式チェック（refs/heads/やrefs/remotes/で始まらない）
98        // ただし、これらは完全に無効ではなく、特別な処理が必要な場合がある
99
100        Ok(())
101    }
102
103    /// ブランチ名を文字列として取得
104    pub fn as_str(&self) -> &str {
105        &self.name
106    }
107
108    /// ブランチ名を所有権付きで取得
109    pub fn into_string(self) -> String {
110        self.name
111    }
112
113    /// デフォルトブランチかどうかを判定
114    pub fn is_default_branch(&self) -> bool {
115        matches!(
116            self.name.as_str(),
117            "main" | "master" | "develop" | "development"
118        )
119    }
120
121    /// リリースブランチかどうかを判定
122    pub fn is_release_branch(&self) -> bool {
123        let release_patterns = [r"^release/.*", r"^releases/.*", r"^rel/.*", r"^v\d+\.\d+.*"];
124
125        release_patterns
126            .iter()
127            .any(|pattern| Regex::new(pattern).unwrap().is_match(&self.name))
128    }
129
130    /// フィーチャーブランチかどうかを判定
131    pub fn is_feature_branch(&self) -> bool {
132        let feature_patterns = [r"^feature/.*", r"^feat/.*", r"^features/.*"];
133
134        feature_patterns
135            .iter()
136            .any(|pattern| Regex::new(pattern).unwrap().is_match(&self.name))
137    }
138
139    /// ホットフィックスブランチかどうかを判定
140    pub fn is_hotfix_branch(&self) -> bool {
141        let hotfix_patterns = [r"^hotfix/.*", r"^hotfixes/.*", r"^fix/.*"];
142
143        hotfix_patterns
144            .iter()
145            .any(|pattern| Regex::new(pattern).unwrap().is_match(&self.name))
146    }
147
148    /// ブランチの種類を取得
149    pub fn branch_type(&self) -> BranchType {
150        if self.is_default_branch() {
151            BranchType::Default
152        } else if self.is_release_branch() {
153            BranchType::Release
154        } else if self.is_feature_branch() {
155            BranchType::Feature
156        } else if self.is_hotfix_branch() {
157            BranchType::Hotfix
158        } else {
159            BranchType::Other
160        }
161    }
162}
