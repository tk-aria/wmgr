3/// リモートリポジトリの情報
4#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
5pub struct Remote {
6    /// リモート名（例: origin）
7    pub name: String,
8    /// リモートのURL
9    pub url: String,
10}
11
12impl Remote {
13    /// 新しいRemoteインスタンスを作成
14    pub fn new(name: impl Into<String>, url: impl Into<String>) -> Self {
15        Self {
16            name: name.into(),
17            url: url.into(),
18        }
19    }
20}
21
22/// リポジトリエンティティ
23#[derive(Debug, Clone, Serialize, Deserialize)]
24pub struct Repository {
25    /// ワークスペース内でのリポジトリの相対パス
26    pub dest: String,
27
28    /// リモートリポジトリのリスト
29    pub remotes: Vec<Remote>,
30
31    /// 対象ブランチ名
32    pub branch: Option<String>,
33
34    /// 元のブランチ名（同期前のブランチ）
35    pub orig_branch: Option<String>,
36
37    /// ブランチを維持するかどうか
38    pub keep_branch: bool,
39
40    /// デフォルトブランチかどうか
41    pub is_default_branch: bool,
42
43    /// 固定されたSHA1ハッシュ
44    pub sha1: Option<String>,
45
46    /// 完全なSHA1ハッシュ
47    pub sha1_full: Option<String>,
48
49    /// タグ名
50    pub tag: Option<String>,
51
52    /// shallow cloneを使用するか
53    pub shallow: bool,
54
55    /// ベアリポジトリかどうか
56    pub is_bare: bool,
57}
58
59impl Repository {
60    /// 新しいRepositoryインスタンスを作成
61    pub fn new(dest: impl Into<String>, remotes: Vec<Remote>) -> Self {
62        Self {
63            dest: dest.into(),
64            remotes,
65            branch: None,
66            orig_branch: None,
67            keep_branch: false,
68            is_default_branch: false,
69            sha1: None,
70            sha1_full: None,
71            tag: None,
72            shallow: false,
73            is_bare: false,
74        }
75    }
76
77    /// ブランチを設定
78    pub fn with_branch(mut self, branch: impl Into<String>) -> Self {
79        self.branch = Some(branch.into());
80        self
81    }
82
83    /// SHA1を設定
84    pub fn with_sha1(mut self, sha1: impl Into<String>) -> Self {
85        self.sha1 = Some(sha1.into());
86        self
87    }
88
89    /// タグを設定
90    pub fn with_tag(mut self, tag: impl Into<String>) -> Self {
91        self.tag = Some(tag.into());
92        self
93    }
94
95    /// shallow cloneを有効化
96    pub fn with_shallow(mut self, shallow: bool) -> Self {
97        self.shallow = shallow;
98        self
99    }
100
101    /// クローン用のURLを取得（最初のリモートのURL）
102    pub fn clone_url(&self) -> Option<&str> {
103        self.remotes.first().map(|r| r.url.as_str())
104    }
105
106    /// 特定の名前のリモートを取得
107    pub fn get_remote(&self, name: &str) -> Option<&Remote> {
108        self.remotes.iter().find(|r| r.name == name)
109    }
110
111    /// デフォルトのリモート（origin）を取得
112    pub fn get_origin(&self) -> Option<&Remote> {
113        self.get_remote("origin")
114    }
115
116    /// リポジトリが固定参照（SHA1またはタグ）を持っているか
117    pub fn has_fixed_ref(&self) -> bool {
118        self.sha1.is_some() || self.tag.is_some()
119    }
120}

122#[cfg(test)]
123mod tests {
124    use super::*;
125
126    #[test]
127    fn test_remote_creation() {
128        let remote = Remote::new("origin", "git@github.com:example/repo.git");
129        assert_eq!(remote.name, "origin");
130        assert_eq!(remote.url, "git@github.com:example/repo.git");
131    }
132
133    #[test]
134    fn test_repository_creation() {
135        let remotes = vec![Remote::new("origin", "git@github.com:example/repo.git")];
136        let repo = Repository::new("path/to/repo", remotes);
137
138        assert_eq!(repo.dest, "path/to/repo");
139        assert_eq!(repo.remotes.len(), 1);
140        assert_eq!(repo.clone_url(), Some("git@github.com:example/repo.git"));
141    }
142
143    #[test]
144    fn test_repository_builder() {
145        let remotes = vec![Remote::new("origin", "git@github.com:example/repo.git")];
146        let repo = Repository::new("path/to/repo", remotes)
147            .with_branch("main")
148            .with_sha1("abc123")
149            .with_shallow(true);
150
151        assert_eq!(repo.branch, Some("main".to_string()));
152        assert_eq!(repo.sha1, Some("abc123".to_string()));
153        assert!(repo.shallow);
154        assert!(repo.has_fixed_ref());
155    }
156
157    #[test]
158    fn test_get_remote() {
159        let remotes = vec![
160            Remote::new("origin", "git@github.com:example/repo.git"),
161            Remote::new("upstream", "git@github.com:upstream/repo.git"),
162        ];
163        let repo = Repository::new("path/to/repo", remotes);
164
165        assert!(repo.get_remote("origin").is_some());
166        assert!(repo.get_remote("upstream").is_some());
167        assert!(repo.get_remote("nonexistent").is_none());
168
169        let origin = repo.get_origin().unwrap();
170        assert_eq!(origin.name, "origin");
171    }
